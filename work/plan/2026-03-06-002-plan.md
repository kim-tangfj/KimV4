## 一、设计原则

### ⚠️ 核心约束
1. **现有页面布局必须保持不变** - 四栏横向布局（项目/片段/镜头/提示词）+ 底部属性面板
2. **素材库窗体独立存在** - 不占用主窗体空间
3. **向外吸附展开** - 向左侧展开，不遮挡主窗体内容

## 六、JavaScript 功能实现

### 6.1 素材按钮点击事件

```javascript
// 缓存素材库侧边窗体元素
const assetsSidebar = document.getElementById('project-assets-sidebar');
const assetsProjectName = document.getElementById('assets-project-name');

// 项目列表项中的素材按钮点击事件
document.querySelectorAll('.assets-btn').forEach(btn => {
  btn.addEventListener('click', (e) => {
    e.stopPropagation();
    const projectId = btn.dataset.projectId;
    openAssetsSidebar(projectId);
  });
});

// 打开素材库侧边窗体
function openAssetsSidebar(projectId) {
  // 查找项目信息
  const project = appState.projects.find(p => p.id === projectId);
  if (!project) return;
  
  // 更新项目信息
  assetsProjectName.textContent = project.name;
  
  // 显示侧边窗体
  assetsSidebar.style.display = 'flex';
  assetsSidebar.classList.remove('hidden');
  
  // 加载素材列表
  loadAssetsList(projectId);
}

// 关闭素材库侧边窗体
function closeAssetsSidebar() {
  assetsSidebar.classList.add('hidden');
  setTimeout(() => {
    assetsSidebar.style.display = 'none';
  }, 300);
}

// 关闭按钮事件
document.querySelector('.assets-sidebar-close').addEventListener('click', closeAssetsSidebar);
```

### 6.2 素材列表加载

```javascript
async function loadAssetsList(projectId) {
  // TODO: 从项目目录读取素材索引文件
  // assets/assets-index.json
  
  const assetsList = document.querySelector('.assets-list');
  assetsList.innerHTML = '';
  
  // 示例数据
  const assets = {
    images: [
      { id: 'asset_img_001', name: 'img_01.jpg', size: '1.2MB', path: 'assets/images/img_01.jpg' },
      { id: 'asset_img_002', name: 'img_02.png', size: '2.3MB', path: 'assets/images/img_02.png' }
    ],
    videos: [
      { id: 'asset_vid_001', name: 'vid_01.mp4', size: '5.2MB', path: 'assets/videos/vid_01.mp4' }
    ],
    audios: [
      { id: 'asset_aud_001', name: 'aud_01.mp3', size: '3.1MB', path: 'assets/audios/aud_01.mp3' }
    ]
  };
  
  // 渲染素材列表
  renderAssetsList(assets);
}

function renderAssetsList(assets) {
  const assetsList = document.querySelector('.assets-list');
  
  // 渲染图片
  if (assets.images.length > 0) {
    assetsList.innerHTML += `
      <div class="assets-section-title">图片</div>
      <div class="assets-grid assets-grid-images">
        ${assets.images.map(asset => `
          <div class="asset-thumbnail" data-asset-id="${asset.id}" data-asset-type="image">
            <img src="${asset.path}" alt="${asset.name}" />
            <div class="asset-info">
              <span class="asset-name">${asset.name}</span>
              <span class="asset-size">${asset.size}</span>
            </div>
          </div>
        `).join('')}
      </div>
    `;
  }
  
  // 渲染视频
  if (assets.videos.length > 0) {
    assetsList.innerHTML += `
      <div class="assets-section-title">视频</div>
      <div class="assets-grid assets-grid-videos">
        ${assets.videos.map(asset => `
          <div class="asset-thumbnail" data-asset-id="${asset.id}" data-asset-type="video">
            <div class="video-thumbnail">
              <span class="video-icon">🎬</span>
            </div>
            <div class="asset-info">
              <span class="asset-name">${asset.name}</span>
              <span class="asset-size">${asset.size}</span>
            </div>
          </div>
        `).join('')}
      </div>
    `;
  }
  
  // 渲染音频
  if (assets.audios.length > 0) {
    assetsList.innerHTML += `
      <div class="assets-section-title">音频</div>
      <div class="assets-grid assets-grid-audios">
        ${assets.audios.map(asset => `
          <div class="asset-thumbnail" data-asset-id="${asset.id}" data-asset-type="audio">
            <div class="audio-thumbnail">
              <span class="audio-icon">🎵</span>
            </div>
            <div class="asset-info">
              <span class="asset-name">${asset.name}</span>
              <span class="asset-size">${asset.size}</span>
            </div>
          </div>
        `).join('')}
      </div>
    `;
  }
}
```

## 九、注意事项

### ⚠️ 核心约束
1. **不修改主窗体布局** - 四栏布局保持不变
2. **侧边窗体向外展开** - 向左侧展开，不遮挡主窗体

1. 使用 `position: fixed` 定位侧边窗体
2. 使用 `transform: translateX()` 实现展开/收起动画
3. 素材路径使用相对路径，便于项目迁移
4. 删除素材前检查 `usageCount`
